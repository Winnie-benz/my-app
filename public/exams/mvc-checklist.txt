    // Thai-safe canvas text helper
    function hasThai(text) { return /[฀-๿]/.test(text); }

    function wrapForCanvas(text, ctx, maxW) {
        if (!text) return [''];
        var isThai = hasThai(text);
        var tokens = isThai ? Array.from(text) : text.split(' ');
        var lines = [], line = '';
        tokens.forEach(function (tok) {
            var candidate = line ? (isThai ? line + tok : line + ' ' + tok) : tok;
            if (ctx.measureText(candidate).width <= maxW) {
                line = candidate;
            } else {
                if (line) lines.push(line);
                line = tok;
            }
        });
        if (line) lines.push(line);
        return lines;
    }

    function feedbackToDataURL(feedback, opts) {
        opts = opts || {};
        var fontFamily  = opts.fontFamily  || '"Sarabun","Inter",sans-serif';
        var fontPx      = opts.fontPx      || 16;
        var lineHeight  = opts.lineHeight  || 1.4;
        var padding     = opts.padding     || 8;
        var maxWidthPx  = opts.maxWidthPx  || 800;
        var textColor   = opts.textColor   || '#444';
        var bg          = opts.bg          || '#ffffff';
        var scale       = opts.scale       || 2;

        var off = document.createElement('canvas');
        var ctx = off.getContext('2d');
        ctx.font = fontPx + 'px ' + fontFamily;
        var lines       = wrapForCanvas(feedback, ctx, maxWidthPx - padding * 2);
        var lineHeightPx = Math.round(fontPx * lineHeight);
        var w = maxWidthPx, h = padding * 2 + lines.length * lineHeightPx;
        off.width  = w * scale;
        off.height = h * scale;
        var ctx2 = off.getContext('2d');
        ctx2.scale(scale, scale);
        ctx2.fillStyle = bg;
        ctx2.fillRect(0, 0, w, h);
        ctx2.font          = fontPx + 'px ' + fontFamily;
        ctx2.fillStyle     = textColor;
        ctx2.textBaseline  = 'top';
        var y = padding;
        lines.forEach(function (l) { ctx2.fillText(l, padding, y); y += lineHeightPx; });
        return { dataURL: off.toDataURL('image/png'), pixelWidth: w, pixelHeight: h };
    }

    async function generatePDF() {

        var originalText = submitBtn.innerText;

            }

            function checkY(needed) {
                needed = needed || 25;
                if (y < M + needed) {
                    page = pdfDoc.addPage([PAGE_W, PAGE_H]);
                    y    = PAGE_H - M;

            }

            y -= 10;

                        var pdfH    = Math.round(fbData.pixelHeight * (maxW / fbData.pixelWidth));
                        checkY(pdfH + 5);

                    } catch (e) {}
                }

                for (var j = 0; j < images.length; j++) {
                    try {
                        checkY(80);
                        var imgBytes = await fetch(images[j].src).then(function (r) { return r.arrayBuffer(); });

                            ? await pdfDoc.embedPng(imgBytes)
                            : await pdfDoc.embedJpg(imgBytes);
                        var iW = 140, iH = embImg.height * (140 / embImg.width);
                        page.drawImage(embImg, { x: M + 4, y: y - iH, width: iW, height: iH });
                        y -= iH + 5;
                    } catch (e) {}
                }
                y -= 6;
            }

            checkY(160);
            y -= 8;
            drawText('Signatures', M, y, { size: 12, f: fontBold });
            y -= 18;
            var sigW = (PAGE_W - M * 2 - 15) / 2, sigH = 100;

            page.drawRectangle({ x: M, y: y - sigH, width: sigW, height: sigH, borderColor: rgb(0.7, 0.7, 0.7), borderWidth: 1 });
            drawText('Staff Signature', M + 4, y - 12, { size: 9, color: rgb(0.5, 0.5, 0.5) });
            if (!staffSignaturePad.isEmpty()) {
                try {
                    var sBytes = await fetch(staffSignaturePad.toDataURL('image/png')).then(function (r) { return r.arrayBuffer(); });
                    var sSig   = await pdfDoc.embedPng(sBytes);
                    page.drawImage(sSig, { x: M + 4, y: y - sigH + 4, width: sigW - 8, height: sigH - 8 });
                } catch (e) {}
            }

            var tx = M + sigW + 15;
            page.drawRectangle({ x: tx, y: y - sigH, width: sigW, height: sigH, borderColor: rgb(0.7, 0.7, 0.7), borderWidth: 1 });
            drawText('Trainer Signature', tx + 4, y - 12, { size: 9, color: rgb(0.5, 0.5, 0.5) });
            if (!trainerSignaturePad.isEmpty()) {
                try {
                    var tBytes = await fetch(trainerSignaturePad.toDataURL('image/png')).then(function (r) { return r.arrayBuffer(); });
                    var tSig   = await pdfDoc.embedPng(tBytes);
                    page.drawImage(tSig, { x: tx + 4, y: y - sigH + 4, width: sigW - 8, height: sigH - 8 });
                } catch (e) {}
            }

            var pdfBytes = await pdfDoc.save();
            var blob = new Blob([pdfBytes], { type: 'application/pdf' });
            var url  = URL.createObjectURL(blob);
            var a    = document.createElement('a');
            a.href     = url;
            a.download = 'Checklist_' + staffId + '_' + examDate + '.pdf';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);

        } catch (err) {
            alert('เกิดข้อผิดพลาดในการสร้าง PDF: ' + err.message);
            console.error(err);
        } finally {
            submitBtn.disabled  = false;
            submitBtn.innerText = originalText;
        }
    }

    // Submit
    submitBtn.addEventListener('click', async function () {
        await generatePDF();
        sendToSheet();
    });

});
