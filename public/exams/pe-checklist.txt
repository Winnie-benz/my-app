function updateSectionScores() \{\
  ['brandImage', 'serviceExperience'].forEach(key => \{\
    const items = checklist[key].items;\
    const scoreEl = document.getElementById('secScore_' + key);\
    const scored = items.filter(it => scores[it.id] && scores[it.id].score !== undefined);\
    if (scored.length === 0) \{ scoreEl.textContent = '\'97 %'; return; \}\

    if (key === 'brandImage') \{\
      const applicable = scored.filter(it => scores[it.id].score !== 'N/A');\
      const passed = applicable.filter(it => scores[it.id].score === 'Pass');\
      const rate = applicable.length > 0 ? Math.round((passed.length / applicable.length) * 100) : 0;\
      scoreEl.textContent = rate + '%';\
      applyScoreStyle(scoreEl, rate);\
    \} else \{\
      const weights = GRADE_WEIGHTS[selectedStaff.grade] || DEFAULT_WEIGHTS;\
      const critical = scored.filter(it => it.type === 'passfail');\
      const nonCrit = scored.filter(it => it.type === 'scale');\
      const critApplicable = critical.filter(it => scores[it.id].score !== 'N/A');\
      const critPass = critApplicable.filter(it => scores[it.id].score === 'Pass');\
      const critRate = critApplicable.length > 0 ? (critPass.length / critApplicable.length) * 100 : 0;\
      const ncScores = nonCrit.map(it => Number(scores[it.id].score) || 0);\
      const ncAvg = ncScores.length > 0 ? (ncScores.reduce((a,b) => a+b, 0) / (ncScores.length * 2)) * 100 : 0;\
      const total = Math.round((critRate * weights.critical) + (ncAvg * weights.nonCritical));\
