        function wrapForCanvas(text, ctx, maxWidthPx) \{\
            if (!text) return [''];\
            const isThai = hasThai(text);\
            const tokens = isThai ? Array.from(text) : text.split(' ');\
            const lines = [];\
            let line = '';\
            for (const tok of tokens) \{\
                const candidate = line ? (isThai ? line + tok : line + ' ' + tok) : tok;\
                if (ctx.measureText(candidate).width <= maxWidthPx) \{\
                    line = candidate;\
                \} else \{\
                    if (line) lines.push(line);\
                    line = tok;\
                    if (ctx.measureText(line).width > maxWidthPx) \{\
                        let buff = '';\
                        for (const ch of Array.from(tok)) \{\
                            if (ctx.measureText(buff + ch).width <= maxWidthPx) \{\
                                buff += ch;\
                            \} else \{\
                                if (buff) lines.push(buff);\
                                buff = ch;\
                            \}\
                        \}\
                        line = buff;\
                    \}\
                \}\
            \}\
            if (line) lines.push(line);\
            return lines;\
        \}\

                            \} else \{\

                                currentLine = word;\
                            \}\

                        lines.push(currentLine);\
                    \});\
